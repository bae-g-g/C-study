# Bitwise

## **Bitwise Operator 비트 연산자**

Bit 변환 연산자
모든 Data는 Byte(=Bit 구성) 단위로 저장 → Bit 값 변환 → Byte 값 변환

| & | 좌항값과 우항값의 Bit가 모두 1 → 1 아니면 0 | A & B |
| ---| ---| --- |
| | | 좌항값과 우항값의 Bit중 하나라도 1 → 1 아니면 0 | A | B |
| ^ | 좌항값과 우항값의 Bit가 다르면 1 → 1 아니면 0 | A ^ B |
| ~ | Bit 0 → 1, 1 → 0 변환 | A ~ B |
| << | 좌항값의 Bit를 우항값 크기만큼 왼쪽 이동 | A << B |
| \>> | 좌항값의 Bit를 우항값 크기만큼 오른쪽 이동 | A >> B |

**bit\_1.c**

```cpp
#include <stdio.h>

int main()
{
    // 8 Bit만 입력위해 char 문자자료형 메모리박스 선언
    char A = 0b00000000;   // 0b → 2진수 Bit 입력/저장
    char B = 0b11111111;   
    char C = 0b00000001;   
    char D = 0b11111110;  

    printf("%d\n", A);     // 10진수 0
    printf("%d\n", B);     // 2진수 11111111 → 맨 처음 1 = 음수  → 2의보수 계산 10진수 -1
    printf("%d\n", C);     // 10진수 1
    printf("%d\n", D);     // 2진수 11111110 → 맨 처음 1 = 음수  → 2의보수 계산 10진수 -2
    printf("=========\n");


    printf("%d\n", C & D); // Bit 각 자릿수 비교: 모두 1이면 1 아니면 0 → 00000000
    printf("%d\n", A);     // 00000000
    printf("=========\n");


    printf("%d\n", C | D); // Bit 각 자릿수 비교: Bit 중 하나라도 1이면 1 아니면 0 → 11111111
    printf("%d\n", B);     // 11111111
    printf("=========\n");

    printf("%d\n", C ^ D); // Bit 각 자릿수 비교: 두 Bit 가 서로 다르면 1 같으면 0 → 11111111
    printf("%d\n", B);     // 11111111
    printf("=========\n");

    printf("%d\n", ~C);    // Bit  1은 0, 0은 1 변환 → 11111110
    printf("%d\n", D);     // 11111110
    printf("=========\n");

    return 0;
}


```

**bit\_2.c**

```cpp
#include <stdio.h>

int main()
{
    char A = 0b00000001;   // 0b → 2진수 Bit 입력/저장
    char B = 0b11111110; 
    char C = 0b00000000; 
    char D = 0b11111111; 

    printf("%d\n", A);          // 1
    printf("%d\n", B);          // 2진수 11111110 → 맨 처음 1 = 음수  → 2의보수 계산 10진수 -2
    printf("%d\n", C);          // 0
    printf("%d\n", D);          // 2진수 11111111 → 맨 처음 1 = 음수  → 2의보수 계산 10진수 -1
    printf("=========\n");

    // char A, B, C, D 선언 → 변수 A, B, C, D 출력
    printf("%d\n", ~A);         // Bit  1은 0, 0은 1 변환 → 11111110 = B = -2
    printf("%d\n", ~B);         // Bit  1은 0, 0은 1 변환 → 00000001 = A = 1
    printf("%d\n", ~C);         // Bit  1은 0, 0은 1 변환 → 11111111 = D = -1
    printf("%d\n", ~D);         // Bit  1은 0, 0은 1 변환 → 00000000 = C = 0
    printf("=========\n");

    // char A, B, C, D 선언 → 변수 A, B, C, D 출력 X  
    printf("%d\n", 0b11111110); // %d(4 Bytes = 32 Bit) → 0b11111110 = 00000000 00000000 00000000 11111110
    printf("%d\n", 0b00000001); // %d(4 Bytes = 32 Bit) → 0b00000001 = 00000000 00000000 00000000 00000001
    printf("%d\n", 0b11111111); // %d(4 Bytes = 32 Bit) → 0b11111111 = 00000000 00000000 00000000 11111111
    printf("%d\n", 0b00000000); // %d(4 Bytes = 32 Bit) → 0b00000000 = 00000000 00000000 00000000 00000000
    printf("=========\n");

    // char A, B, C, D 선언 → 변수 A, B, C, D 출력값과 동일한 2진수
    printf("%d\n", 0b11111111111111111111111111111110); // 10진수 음수값(=-2): 1로 나머지 24 Bits
                                                        // 원리
                                                        // 2의보수 계산: X + (-X) = 0에서 -X의 10진수 값(=-2)을 알때 -X의 2진수 = X(=2)의 2진수 Not 연산 + 1
                                                        // 10진수 2 = 2진수 00000000 00000000 00000000 00000010
                                                        // 2진수 00000000 00000000 00000000 00000010 Not 연산 + 1
                                                        // 11111111111111111111111111111110       
    printf("%d\n", 0b00000000000000000000000000000001); // 10진수 양수값(= 1): 0로 나머지 24 Bits 
    printf("%d\n", 0b11111111111111111111111111111111); // 10진수 음수값(=-1): 1로 나머지 24 Bits
                                                        // 원리
                                                        // 2의보수 계산: X + (-X) = 0에서 -X의 10진수 값(=-1)을 알때 -X의 2진수 = X(=1)의 2진수 Not 연산 + 1
                                                        // 10진수 2 = 2진수 00000000 00000000 00000000 00000001
                                                        // 2진수 00000000 00000000 00000000 00000001 Not 연산 + 1
                                                        // 11111111111111111111111111111111    
    printf("%d\n", 0b00000000000000000000000000000000); // 10진수 양수값(= 0): 0로 나머지 24 Bits 

    return 0;
}
```

<< >>
*   32 Bit 초과 이동 → 이동크기 % 32 만큼 이동

E.g)

40 크기 이동 → 40 % 32 = 8 만큼 이동

72 크기 이동 → 72 % 32 = 8 만큼 이동

*   이동 후 32 Bit 초과 Bit 버림

**bit\_3.c**

```cpp
#include <stdio.h>

int main()
{
    // Case 1
    int X = 1;          // 10진수 1 = 2진수 00000000000000000000000000000001
    X = X << 2;         // X 변수값(=1) 2진수 Bit 왼쪽으로 2만큼 이동
                        // 00000000000000000000000000000001 → 00000000000000000000000000000100 = 10진수 4
    printf("%d\n", X);  
    
    X = X << 10;        // X 변수값(=4) 2진수 Bit 왼쪽으로 10만큼 이동
                        // 00000000000000000000000000000100 → 00000000000000000001000000000000 = 10진수 4096
    printf("%d\n", X);
    
    X = X << 40;        // X 변수값(=4096) 2진수 Bit 왼쪽으로 40만큼 이동
                        // Compiler 32 Bit 크기만 인식 → 40 % 32 = 8 만큼 이동  
                        // 00000000000000000001000000000000 → 00000000000100000000000000000000 = 10진수 1048576
    printf("%d\n", X);

    X = X << 72;        // X 변수값(=1048576) 2진수 Bit 왼쪽으로 72만큼 이동
                        // Compiler 32 Bit 크기만 인식 → 72 % 32 = 8 만큼 이동  
                        // 00000000000100000000000000000000 → 00010000000000000000000000000000 = 10진수 268435456
    printf("%d\n", X);

    X = X << 3;         // X 변수값(=268435456) 2진수 Bit 왼쪽으로 3만큼 이동
                        // 00010000000000000000000000000000 → 10000000000000000000000000000000
                        // 처음 1 음수 → 2의보수 계산: X + (-X) = 0에서 -X의 2진수를 알때 X = -X의 2진수 Not 연산 + 1의 역연산 = -X의 2진수 -1의 Not 연산
                        // 10000000000000000000000000000000 = 2147483648 = X
                        // -X = -2147483648 
    printf("%d\n", X);
    
    X = X << 3;         // X 변수값(=-2147483648) 2진수 Bit 왼쪽으로 3만큼 이동
                        // 10000000000000000000000000000000 → 00000000000000000000000000000000 = 10진수 0
    printf("%d\n", X);

    // Case 2
    int Y = 11;         // 10진수 11 = 2진수 00000000000000000000000000000001  
    Y = Y >> 1;         // Y 변수값(=11) 2진수 Bit 오른쪽으로 1만큼 이동    
                        // 00000000000000000000000000001011 → 00000000000000000000000000000101(초과 Bit 버림) = 10진수 5
    printf("%d\n", Y);

    Y = Y >> 1;         // Y 변수값(=11) 2진수 Bit 오른쪽으로 1만큼 이동    
                        // 00000000000000000000000000000101 → 00000000000000000000000000000010(초과 Bit 버림) = 10진수 2
    printf("%d\n", Y);

    Y = Y >> 1;         // Y 변수값(=11) 2진수 Bit 오른쪽으로 1만큼 이동    
                        // 00000000000000000000000000000010 → 00000000000000000000000000000001(초과 Bit 버림) = 10진수 1
    printf("%d\n", Y); 

    Y = Y >> 1;         // Y 변수값(=11) 2진수 Bit 오른쪽으로 1만큼 이동    
                        // 00000000000000000000000000000001 → 00000000000000000000000000000000(초과 Bit 버림) = 10진수 0
    printf("%d\n", Y);      

    return 0;
}
```

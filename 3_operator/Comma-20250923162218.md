# Comma

## **Comma Operator 쉼표 연산자**

**com\_1.c**

```cpp
#include <stdio.h>

int main()
{
    int X, Y, Z;

    // Comma가 Parameter 구별로 사용될 때
    scanf("%d %d %d", &X, &Y, &Z);  // %d 정수자료형 1, 2, 3(4 Bytes) X, Y, Z 주소값(&)을 시작으로 입력/저장
                                    // Window OS
                                    // 1) &Z, &Y, &X 순서(R→L)로 각각의 변수 주소(&)에 메모리박스(+TEMP 생성) 마련 
                                    // 2) "%d %d %d" 순서(L→R)로 변수 입력/저장
                                    // 3) scanf 실행(각각의 주소에 변수 입력/저장된 것 읽고 실행 )
    printf("%d\n", X);
    printf("%d\n", Y);
    printf("%d\n", Z);

    // Comma가 Code 구별로 사용될 때
    X = 10, Y = 20, Z = 30, printf("%d\n", X), printf("%d\n", Y), printf("%d\n", Z); // 순서(L→R)  ∵ Code 구별 

    // Comma가 Code 구별로 사용될 때
    // , = "; + \n"
    X = 10;
    Y = 20;
    Z = 30;
    printf("%d\n", X);
    printf("%d\n", Y);
    printf("%d\n", Z);

    return 0;
}


```

**com\_2.c**

```cpp
#include <stdio.h>

int main()
{
    // Comma가 Parameter 구별로 사용될 때
    int A;
    char B;
    scanf("%d %d", &A, &B); // %d 정수자료형 3, 89(4 Bytes) A, B 주소값(&)을 시작으로 입력/저장
                            // Window OS
                            // 1) &B, &A 순서(R→L)로 각각의 변수 주소(&)에 TEMP 생성/마련
                            // 2) "%d %d" 변수 입력/저장: "%d %d" 순서(L→R)로 각각 위치에 Match되는 A, B 변수 입력/저장 (3 → A,  89 → B)
                            // 3) scanf 실행(각각의 주소에 변수 입력/저장된 것 읽고 실행)
    printf("%d %d\n", A, B);

    printf("-------");

    int C;
    char D;
    scanf("%d %d", &D, &C); // %d 정수자료형 3, 89(4 Bytes) C, D 주소값(&)을 시작으로 입력/저장
                            // Window OS
                            // 1) &C, &D 순서(R→L)로 각각의 변수 주소(&)에 TEMP 생성/마련
                            // 2) "%d %d" 변수 입력/저장: "%d %d" 순서(L→R)로 각각 위치에 Match되는 D, C 변수 입력/저장 (3 → D,  89 → C)
                            // 3) scanf 실행(각각의 주소에 변수 입력/저장된 것 읽고 실행)
    printf("%d %d\n", C, D);    
    
    // Comma가 Code 구별로 사용될 때
    // , = "; + \n"
    int X = 1;
    int Y = 11;

    printf("%d\n", X), printf("%d\n", Y), X = 100, printf("%d\n", X);

    return 0;
}


```

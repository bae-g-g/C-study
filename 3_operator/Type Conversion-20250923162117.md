# Type Conversion

## **Type Conversion 형 변환**

`선언한 자료형 형태에 맞춰 변수값 초기화`

### **Example)**

**cv\_1.c**

```cpp
#include <stdio.h>

int main()
{
// 선언 '동시' 초기화
// Case 1-1: char to int    
char A = 5;        // 정수자료형  5(4 Bytes) → 선언 문자자료형 크기 (1 Byte) 만큼 입력/저장: Little Endian  00000101  (00000000  00000000  00000000)
int B = 10;        // 정수자료형 10(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장: Little Endian  00001010   00000000  00000000  00000000
B = A + B;         // int B = 32 Bytes 크기 정수 5의 1 Byte 크기 정수(=5) + 정수 10
printf("%d\n", B); // 선언한 자료형크기(int B: 4 Bytes) 만큼 정수 1개 출력

// Case 1-2: char to int   
char A = 5;      
int B = 10;      
B = (int)A + B;    // int B = 문자→정수 5 + 정수 10
printf("%d\n", B); // 선언한 자료형크기(int B: 4 Bytes) 만큼 정수 1개 출력

// 선언 '후' 초기화
// Case 2-1: char to int    
char A;
int B;
scanf("%d", &A);   // %d 정수자료형     5(4 Bytes) A 주소값(&)을 시작으로 입력/저장: Little Endian  00000101  00000000  00000000  00000000  
scanf("%d", &B);   // %d 정수자료형    10(4 Bytes) B 주소값(&)을 시작으로 입력/저장: Little Endian  00001010  00000000  00000000  00000000

B = A + B;         // int B = 32 Bytes 크기 정수 5의 char A 메모리박스(1 Byte)해당 정수(=5) + 정수 10
printf("%d\n", B); // 선언한 자료형크기(int B: 4 Bytes) 만큼 정수 1개 출력

// Case 2-2: char to int    
char A;
int B;
scanf("%d", &A);  
scanf("%d", &B); 

B = (int)A + B;     // int B = 문자→정수 5 + 정수 10
printf("%d\n", B); 

return 0;
}


```

```cpp
                 // → X의 2진수 = 11100111 Not 연산 + 1
                 // → X의 2진수 = 00011001 = 10진수 25
                 // ∴ X = 25, -X = -25
int B = 10;      // 정수자료형   10(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장: Little Endian  00001010  00000000  00000000  00000000

char A = 2023;
int B = 10;
B = (int)A + B;  // int B = 정수 -25 + 정수 10
printf("%d", B);

// 선언 '후' 초기화
```

```cpp
#include <stdio.h>

int main()
{
// 선언 '동시' 초기화    
char A = 'X';          // 문자 X = ASCII Code 변환 정수 88
char B = 'Y';          // 문자 Y = ASCII Code 변환 정수 89

char C = 'X' + 'Y';    // char C = 정수 88 + 정수 89의 1 Byte 크기 해당 정수

printf("%d\n", A + B); // 177 = 00000000 00000000 00000000 10110001
                       // A + B의 결과값의 초기화할 자료형이 선언 X → 선언한 자료형크기 고려없이 %d 크기/형태(4 Bytes = 32 Bits 정수)로 출력 
                       // → 가장 처음 0 = 양수 → sign 2진수 계산 방식
printf("%d\n", C);     // 177 = 00000000 00000000 00000000 10110001
                       // 입력:  char C(1 Byte) 크기만큼 입력/저장 →  2진수 10110001(Litte Endian)
                       // 출력:  A + B의 결과값의 초기화할 자료형이 선언 O → 선언한 자료형크기(char C: 1 Byte) 만큼 %d 정수형태로 출력 
                       // → 가장 처음 1 = 음수 → 2의 보수 2진수 계산 방식: X + (-X) = 0에서 -X의 2진수(=10110001)를 알때
                       // → X의 2진수 = -X의 2진수의 Not 연산 + 1의 역연산
                       // → X의 2진수 = 10110001 - 1의 Not 연산
                       // → X의 2진수 = 01001111 = 10진수 79
                       // ∴ X = 79, -X = -79
// 선언 '후' 초기화  
char D;
char E;                
char F;
scanf("%c", &D);       // 문자   X  입력/저장
scanf("%c", &E);       // 문자 엔터 입력/저장
scanf("%c", &F);       // 문자   Y  입력/저장

char G = D + F;

printf("%d\n", D + F);
printf("%d\n", G);

return 0;
}
```

```cpp
#include <stdio.h>
// 배열
// Case 1: char to int: 문자 1개 to 정수 자릿수 1자리
int main()
{
// 선언 '동시' 초기화    
char A[3] = {'0', '2', '7'};

int X = A[1] - '0';        // 문자 2 - 문자 0 = ASCII Code 변환 정수 50 - 48 = 정수 2 
int Y = B[1] - '0';        // 문자 6 - 문자 0 = ASCII Code 변환 정수 54 - 48 = 정수 6

printf("%d\n", X);
printf("%d\n", Y);
printf("%d\n", X + Y);

// 선언 '후' 초기화  
char A[3];
char B[3];
scanf("%s", &A);
scanf("%s", &B);

int X = A[1] - '0';         
int Y = B[1] - '0';

printf("%d\n", X);
printf("%d\n", Y);
printf("%d\n", X + Y);

return 0;
}
```

```cpp
#include <stdio.h>

int main()
{
// 선언 '동시' 초기화
// Case 1-1: int & float    
int A = 8;            // 정수자료형     8(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장
float B = 8.625;      // 소수자료형 8.625(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장
A = A + B;            // int A = 8 + 8.625 = 16.625 → 소수자리 제거 → 정수 16
B = A + B;            // float B = 16 + 8.625 = 24.625 → 소수자리 유지 → 소수 24.62500
printf("%d\n", A);    // 선언한 자료형크기(  int A: 4 Bytes) 만큼 정수 1개 출력
printf("%f\n", B);    // 선언한 자료형크기(float B: 4 Bytes) 만큼 소수 1개 출력

// Case 1-2: int & float   
int A = 8;            // 정수자료형     8(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장
float B = 8.625;      // 소수자료형 8.625(4 Bytes) → 선언 문자자료형 크기(4 Bytes) 만큼 입력/저장
A = A + (int)B;       // int A = 8 + (소수 8.625 → 정수 8) = 정수 16
B = (float)A + B;     // float B = (정수 16 → 소수 16.00000) + 8.625 = 소수 24.62500
printf("%d\n", A);    // 선언한 자료형크기(  int A: 4 Bytes) 만큼 정수 1개 출력
printf("%f\n", B);    // 선언한 자료형크기(float B: 4 Bytes) 만큼 소수 1개 출력

// 선언 '후' 초기화
// Case 2-1: int & float     
int A;            
float B;     
scanf("%d", &A);      // %d 정수자료형     8(4 Bytes) A 주소값(&)을 시작으로 입력/저장 
scanf("%f", &B);      // %f 소수자료형 8.625(4 Bytes) B 주소값(&)을 시작으로 입력/저장 

A = A + B;            
B = A + B;            
printf("%d\n", A);    // 선언한 자료형크기(  int A: 4 Bytes) 만큼 정수 1개 출력
printf("%f\n", B);    // 선언한 자료형크기(float B: 4 Bytes) 만큼 소수 1개 출력

// Case 2-2: int & float     
int A;            
float B;     
scanf("%d", &A); 
scanf("%f", &B); 

A = A + (int)B;       
B = (float)A + B;     
printf("%d\n", A);    
printf("%f\n", B);    

return 0;
}
```

```cpp
#include <stdio.h>

int main()
{
    int A;
    float B;

    A = 8.08;               // A:   int(정수) 메모리박스 → 변수 float(= 8.08) → 변수 정수 처리(=8)
    B = 10.18;              // B: float(소수) 메모리박스 → 변수 float(=10.18) → 변수 소수 처리(=10.18)

    printf("%d\n", A);      // 선언한 자료형크기(4 Bytes) 만큼 정수 1개 8 출력
    printf("%f\n", B);      // 선언한 자료형크기(4 Bytes) 만큼 소수 1개 10.18 출력
    printf("%f\n", A + B);  // 소수 1개 18.18 출력
    printf("%d\n", A + B);  // 소수 1개 18 출력? → -2147483648 출력

    printf("--------------\n");

    A = (float)8.08;        // A:   int(정수) 메모리박스 → 변수 float 형변환( =8.08) → 변수 정수 처리(=8)
    B = (int)10.18;         // B: float(소수) 메모리박스 → 변수   int 형변환(=10.00) → 변수 소수 처리(=10.00)

    printf("%d\n", A);      // 선언한 자료형크기(4 Bytes) 만큼 정수 1개     8 출력
    printf("%f\n", B);      // 선언한 자료형크기(4 Bytes) 만큼 소수 1개 10.00 출력
    printf("%f\n", A + B);  // 소수 1개 18.00 출력
    printf("%d\n", A + B);  // 정수 1개 18 출력? → 0 출력
    
    return 0;
}
```

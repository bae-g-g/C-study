# Return Value 無

## **반환값(Reutrn값) '無' 함수**

**b\_function\_void\_1.c**

```cpp
#include <stdio.h>

void main()
{
    printf("Design Embodiement Module_C Language");
}
```

반환자료형: void
함수명: main
매개변수: 無
반환값: 無(void)

**b\_function\_void\_2.c**

```cpp
#include <stdio.h>

void function_1()    // 함수선언: 반환 자료형 void인 매개변수 0개인 함수 
{                    // 함수정의: 4~7번째줄
    printf("Design Embodiement Module_C Language\n"); 
    return;          // 함수리턴: void 함수경우 return 자체를 작성하지 않아도 가능 O   ∵ 반환(Return)해줄 값 無
}

void main()
{
    function_1();    // 함수호출: 반환자료형이 없는경우의 호출 형식
    printf("함수 function_1 실행종료\n");
}
```

`()` ≠ `(void)`
*   ()

매개변수가 정해지지 않음 (매개변수 자료형이 Default값인 int로 인식)

*   (void)

매개변수가 없음

매개변수 초기화 시도 → Code Error

#include <stdio.h>

void function_1(void)       // 반환자료형 void,  매개변수 = void,  반환값 = void
{
    printf("function_1\n");
}

void function_2()           // 반환자료형 void,  매개변수 ≠ void,  반환값 = void
{
    printf("function_2\n");
}

function_3(void)            // 반환자료형 Default값 = int,  매개변수 = void
{
    printf("function_3\n"); 
}                           // 반환자료형 有 함수 = int: 반환값 작성 X  → 반환값 = 쓰레기값

function_4()                // 반환자료형 Default값 = int,  매개변수 ≠ void, 반환값 ≠ void
{
    printf("function_4\n");
    return 99;              // 반환자료형 有 함수 = int: 반환값 작성 O  → 반환값 = 정수 99
}

void main()                 // 반환자료형 void,  매개변수 ≠ void,  반환값 = void
{
    int X;
                         //  매개변수  반환값
    function_1();        //   void     void
    function_2();        //    ()      void
    function_3();        //   void      ()
    function_4();        //    ()       ()

    // X = function_1(); //   반환값 = void:  반환값이 없는데 X에 초기화 시도 → Error O
    // function_1(10);   // 매개변수 = void:  인수 10을 3번째줄 매개변수에 재초기화 시도 → error O

    // X = function_2(); //   반환값 = void:  반환값이 없는데 X에 초기화 시도 → Error
    function_2(10);      // 매개변수 ≠ void:  인수 10을 8번째줄 매개변수에 재초기화 시도 → error X

    X = function_3();
    function_3(10);      // 매개변수 = void:  인수 10을 13번째줄 매개변수에 재초기화 시도 → error O

    X = function_4();

}

**b\_function\_void\_4.c**

```cpp
#include <stdio.h>

function_1(Y)       // 반환자료형 Default값 = int
{
    return Y;
}

void main()         // 반환자료형 void,  매개변수 ≠ void,  반환값 = void
{
    int X = function_1();
    printf("%d\n", X);               // 5번째줄 반환값(Return값) ≠ 정수형(Y)  → 반환값 = 쓰레기값
    printf("%d\n", function_1());    // 5번째줄 반환값(Return값) ≠ 정수형(Y)  → 반환값 = 쓰레기값
    printf("%d\n", function_1(200)); // 5번째줄 반환값(Return값) = 정수형(200 = Y)  → 반환값 = Y = 200
}
```

// 반환자료형 Default값 = int 확인 Code

#include <stdio.h>

function_1(E)       // 반환자료형 Default값 = int  → 반환값 유형 must int
{
    return E;
}

function_2(float E) // 매개변수 자료형 float 선언,  But 반환자료형 Default값 = int  → 반환값 유형 must int
{
    return E;
}

void main()         // 반환자료형 void,  매개변수 ≠ void,  반환값 = void
{
    int A = function_1(10);
    printf("%d\n", A);
    
    float B = function_1(1.1);    // Window OS 내부적으로 봐볼 필요 있음 / Mac OS 정상 출력
    int B_int = function_1(1.1);  // Window OS 내부적으로 봐볼 필요 있음 / Mac OS 정상 출력
    printf("%f\n", B);
    printf("%d\n", B_int);

    float C = function_2(1.1);
    int D = function_2(1.1);
    printf("%f\n", C);
    printf("%d\n", D);
}

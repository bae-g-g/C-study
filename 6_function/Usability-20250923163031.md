**g\_function\_1.c**
함수사용 X

```cpp
#include <stdio.h>

int main()
{
    int saving_rate = 0.04;   // 저축 이율 = 4%
    float saving_amount;      // 저축 금액 
    int saving_month;         // 저축 유지 개월수

    // 100만원이 12개월 후에 얼마가 되는지 계산
    saving_amount = 100;
    saving_month = 12;
    for (int i = 0; i < saving_month; i++)
    {
        saving_amount = saving_amount * (1 + saving_rate);
    }
    printf("금액 : %f\n", saving_amount);

    // 100만원이 24개월 후에 얼마가 되는지 계산
    saving_amount = 100;
    saving_month = 24;
    for (int i = 0; i < saving_month; i++)
    {
        saving_amount = saving_amount * (2 + saving_rate);  // 식을 잘못 쓴 부분 → 직접 이렇게 Code를 하나하나 쓰다보면 이렇게 실수가 발생(2 → 1로 수정 필요) → 함수였으면 한번만 고치면 되나 일일이 다 고쳐야함
    }
    printf("금액 : %f\n", saving_amount);

    // 70만원이 16개월 후에 얼마가 되는지 계산
    saving_amount = 70;
    saving_month = 16;
    for (int i = 0; i < saving_month; i++)
    {
        saving_amount = saving_amount * (1 + saving_rate);
    }
    printf("금액 : %f\n", saving_amount);

    return 0;
}
```

**g\_function\_2.c**
함수사용 O

```cpp
#include <stdio.h>

// 저축금액과 저축이율로 최종금액 계산
float saving_account(float saving_amount, int saving_month)
{
    float saving_rate = 0.04;
    for (int i = 0; i < saving_month; i++)
    {
        saving_amount = saving_amount * (1 + saving_rate);
    }
    return saving_amount;
}

// Approach 1
int main()
{
    printf("%f\n", saving_account(100, 12)); // 100만원이 12개월 후에 얼마가 되는지 계산
    printf("%f\n", saving_account(100, 24)); // 100만원이 24개월 후에 얼마가 되는지 계산
    printf("%f\n", saving_account(70, 16));  //  70만원이 16개월 후에 얼마가 되는지 계산

    return 0;
}
// Approach 2-1
int main()
{
    int invest_money, invest_month;
    printf("예금액과 예금개월수를 입력하세요.\n");
    scanf("%d", &invest_money);
    scanf("%d", &invest_month);

    printf("최종 예상 금액은 %f원 입니다.\n", saving_account(invest_money, invest_month));  

    return 0;
}
// Approach 2-2
int main()
{
    int invest_money, invest_month;
    printf("예금액과 예금개월수를 입력하세요.\n");
    scanf("%d", &invest_money);
    scanf("%d", &invest_month);

    float final_return = saving_account(invest_money,  invest_month);

    printf("최종금액은 %f입니다.\n", final_return);  

    return 0;
}
```

이자율(saving\_rate) 변경시
함수사용 X → 해당 여러 Codes 수정 要
함수사용 O → 해당 1개 함수 수정 要
